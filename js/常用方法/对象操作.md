## 对象循环赋值

```javascript
let arrA={
    name:'',
    age:'',
    sex:'',
}

let arrB={
    name:'charlie',
    age:'11'
}
const listAssign = (arrA, arrB) => Object.keys(arrA).forEach(key => { arrA[key] = arrB[key] || arrA[key]});
listAssign(arrA,arrB)

console.log(arrA);

```
## 对象属性置空
```javascript
let obj={
    name:'charlie',
    age:'24',
    sex:'man'
}
function empty(obj={}){
    for (let i in obj) {
        obj[i]=null
    }
    return obj
}
```
