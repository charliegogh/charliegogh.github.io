# wepy的使用

# 全局安装

npm install wepy-cli -g

# 安装wepy-cli

npm install -g wepy-cli
wepy init standard my-project

# 切换到目录

cd xxx

# 安装依赖

npm i

# 开启实时编译

wepy build --watch
